    return price * (1 + tax_percent / 100)

def calculate_final_price(
    price: float,
    discount_percent: float,
    tax_percent: float
) -> float:
    discounted_price = apply_discount(price, discount_percent)
    final_price = apply_tax(discounted_price, tax_percent)
